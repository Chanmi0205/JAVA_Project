
public class Student {
	
	// 이름
	public String name;
	
	// 나이
	public int ige;
	
	// 성별
	char gender;
	
	// 키
	double height;
	
	// 생년월일
	int birthDate;
	
	// 생성자
	public Student() {
		System.out.println("디폴트 생성자");
	}
	
	// 매개변수(name, ige, gender, height, birthDate) 받는 생성자
	public Student(String name, int ige, char gender, double height, int birthDate) {
		this.name = name;
		this.ige = ige;
		this.gender = gender;
		this.height = height;
		this.birthDate = birthDate;
	}
	
	public Student(String name, int ige) {
		this.name = name;
		this.ige = ige;
	}

	// 생각하다
	public void think() {
		System.out.println("닭강정 먹고싶다...");
	}
	
	// 숨쉬다
	public void huhahuha() {
		System.out.println("후하후하");
	}
}
